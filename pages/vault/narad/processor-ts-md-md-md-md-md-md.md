import { SupabaseClient } from '@supabase/supabase-js';
import { calculateMatch, UserProfile, MatchResult } from '@/lib/matcher';
import { saveJob, UnifiedJob } from './dedupe';

export interface MatchedJob extends UnifiedJob {
  matchScore: number;
  recommendation: MatchResult['recommendation'];
}

export interface JobProcessorOptions {
  userId: string;
  userProfile: UserProfile;
  matchThreshold?: number;
  saveAll?: boolean;
  supabase: SupabaseClient;
}

function unifyToJobListing(job: UnifiedJob) {
  const skillsMatch = job.description?.match(/required skills?[:\s]*([^\n.]+)/i);
  const skills = skillsMatch
    ? skillsMatch[1].split(/[,;]/).map((s: string) => s.trim()).filter(Boolean)
    : [];

  return {
    title: job.title,
    description: job.description,
    requiredSkills: skills,
    location: job.location,
    remote: job.remote,
  };
}

export async function processAndMatchJobs(
  jobs: UnifiedJob[],
  options: JobProcessorOptions
): Promise<{
  jobs: MatchedJob[];
  saved: number;
  skipped: number;
}> {
  const { userId, userProfile, matchThreshold = 70, saveAll = false, supabase } = options;

  const matchedJobs: MatchedJob[] = [];
  let saved = 0;
  let skipped = 0;

  const jobListings = jobs.map(job => ({
    ...unifyToJobListing(job),
    requiredSkills: [],
  }));

  for (let i = 0; i < jobs.length; i++) {
    const job = jobs[i];
    const jobListing = {
      title: job.title,
      description: job.description,
      requiredSkills: extractSkills(job.description),
      location: job.location,
    };

    const matchResult = calculateMatch(userProfile, jobListing);

    const matchedJob: MatchedJob = {
      ...job,
      matchScore: matchResult.overallScore,
      recommendation: matchResult.recommendation,
    };

    const meetsThreshold = matchResult.overallScore >= matchThreshold;

    if (meetsThreshold || saveAll) {
      try {
        await saveJob(supabase, job, userId);
        saved++;
      } catch (error) {
        console.error('Error saving job:', job.title, error);
        skipped++;
      }
    } else {
      skipped++;
    }

    if (meetsThreshold || saveAll) {
      matchedJobs.push(matchedJob);
    }
  }

  matchedJobs.sort((a, b) => b.matchScore - a.matchScore);

  return { jobs: matchedJobs, saved, skipped };
}

function extractSkills(description: string): string[] {
  const techKeywords = [
    'javascript', 'typescript', 'python', 'java', 'react', 'angular', 'vue',
    'node', 'express', 'django', 'flask', 'spring', 'aws', 'gcp', 'azure',
    'docker', 'kubernetes', 'sql', 'nosql', 'mongodb', 'postgresql', 'mysql',
    'graphql', 'rest', 'api', 'git', 'linux', 'html', 'css', 'tailwind',
    'nextjs', 'svelte', 'rust', 'go', 'golang', 'c++', 'c#', '.net',
    'ruby', 'rails', 'php', 'laravel', 'swift', 'kotlin', 'flutter',
    'machine learning', 'ai', 'data science', 'tensorflow', 'pytorch',
  ];

  const text = description.toLowerCase();
  return techKeywords.filter(skill => text.includes(skill));
}

export async function processJobsBatch(
  jobs: UnifiedJob[],
  options: JobProcessorOptions,
  batchSize = 10
): Promise<{
  jobs: MatchedJob[];
  saved: number;
  skipped: number;
}> {
  const allMatchedJobs: MatchedJob[] = [];
  let totalSaved = 0;
  let totalSkipped = 0;

  for (let i = 0; i < jobs.length; i += batchSize) {
    const batch = jobs.slice(i, i + batchSize);
    const result = await processAndMatchJobs(batch, {
      ...options,
      supabase: options.supabase,
    });

    allMatchedJobs.push(...result.jobs);
    totalSaved += result.saved;
    totalSkipped += result.skipped;
  }

  return { jobs: allMatchedJobs, saved: totalSaved, skipped: totalSkipped };
}
