import { getStore } from './memory.js';

const VERIFICATION_CONFIG = {
  truthThreshold: 0.95,
  minConfidence: 0.7,
  autoVerify: false,
  verificationProviders: ['groq', 'openrouter'],
  maxRetries: 3
};

export async function verifyTruth(env, query, result, options = {}) {
  const threshold = options.threshold || VERIFICATION_CONFIG.truthThreshold;
  const provider = options.provider || 'groq';
  
  if (!result || !query) {
    return { verified: false, confidence: 0, error: 'Missing query or result' };
  }
  
  const verificationPrompt = `
You are a fact verification system. Evaluate the accuracy of the following answer to the query.

QUERY: ${query}

ANSWER: ${result}

Evaluate:
1. Factual accuracy (0-1)
2. Consistency with known information (0-1)  
3. Completeness (0-1)
4. Logical coherence (0-1)

Provide a JSON response with:
{
  "verified": boolean,
  "confidence": number (0-1),
  "accuracy": number (0-1),
  "issues": [string],
  "explanation": string
}

Be strict but fair. Only mark as verified if the answer is demonstrably correct.
`.trim();
  
  let verificationResult;
  
  try {
    if (provider === 'groq' && env.GROQ_API_KEY) {
      verificationResult = await verifyWithGroq(env, verificationPrompt);
    } else if (provider === 'openrouter' && env.OPENROUTER_API_KEY) {
      verificationResult = await verifyWithOpenRouter(env, verificationPrompt);
    } else {
      return { verified: false, confidence: 0, error: 'No verification provider available' };
    }
  } catch (e) {
    return { verified: false, confidence: 0, error: e.message };
  }
  
  await saveVerification(env, query, result, verificationResult);
  
  return {
    ...verificationResult,
    meetsThreshold: verificationResult.confidence >= threshold,
    threshold
  };
}

async function verifyWithGroq(env, prompt) {
  const response = await fetch('https://api.groq.com/openai/v1/chat/completions', {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${env.GROQ_API_KEY}`,
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({
      model: 'llama-3.1-8b-instant',
      messages: [
        { role: 'system', content: 'You are a strict fact verification AI. Respond with ONLY valid JSON.' },
        { role: 'user', content: prompt }
      ],
      temperature: 0.1,
      max_tokens: 500
    })
  });
  
  if (!response.ok) {
    throw new Error(`Groq error: ${response.status}`);
  }
  
  const data = await response.json();
  const content = data.choices?.[0]?.message?.content || '';
  
  return parseVerificationResponse(content);
}

async function verifyWithOpenRouter(env, prompt) {
  const response = await fetch('https://openrouter.ai/api/v1/chat/completions', {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${env.OPENROUTER_API_KEY}`,
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({
      model: 'meta-llama/llama-3.1-8b-instant',
      messages: [
        { role: 'system', content: 'You are a strict fact verification AI. Respond with ONLY valid JSON.' },
        { role: 'user', content: prompt }
      ],
      temperature: 0.1,
      max_tokens: 500
    })
  });
  
  if (!response.ok) {
    throw new Error(`OpenRouter error: ${response.status}`);
  }
  
  const data = await response.json();
  const content = data.choices?.[0]?.message?.content || '';
  
  return parseVerificationResponse(content);
}

function parseVerificationResponse(content) {
  try {
    const jsonMatch = content.match(/\{[\s\S]*\}/);
    if (jsonMatch) {
      return JSON.parse(jsonMatch[0]);
    }
    throw new Error('No JSON found');
  } catch (e) {
    return {
      verified: false,
      confidence: 0.5,
      accuracy: 0.5,
      issues: ['Failed to parse verification response'],
      explanation: 'Verification system error'
    };
  }
}

export async function saveVerification(env, query, result, verification) {
  const tableName = 'verifications';
  const dataKey = `memory:data:${tableName}`;
  const data = await getStore(env).get(dataKey);
  const rows = data ? JSON.parse(data) : [];
  
  const record = {
    id: `verif:${Date.now()}:${Math.random().toString(36).substr(2, 9)}`,
    query,
    result,
    confidence: verification.confidence,
    accuracy: verification.accuracy,
    verified: verification.verified,
    issues: verification.issues || [],
    explanation: verification.explanation,
    verified_at: new Date().toISOString()
  };
  
  rows.push(record);
  
  if (rows.length > 1000) {
    rows.shift();
  }
  
  await getStore(env).put(dataKey, JSON.stringify(rows));
  
  return record;
}

export async function getVerificationHistory(env, limit = 50) {
  const tableName = 'verifications';
  const dataKey = `memory:data:${tableName}`;
  const data = await getStore(env).get(dataKey);
  const rows = data ? JSON.parse(data) : [];
  
  return rows
    .sort((a, b) => new Date(b.verified_at) - new Date(a.verified_at))
    .slice(0, limit);
}

export async function getVerificationStats(env) {
  const history = await getVerificationHistory(env, 1000);
  
  if (history.length === 0) {
    return {
      total: 0,
      verified: 0,
      rejected: 0,
      averageConfidence: 0,
      accuracy: 0
    };
  }
  
  const verified = history.filter(h => h.verified).length;
  const totalConfidence = history.reduce((sum, h) => sum + (h.confidence || 0), 0);
  const totalAccuracy = history.reduce((sum, h) => sum + (h.accuracy || 0), 0);
  
  return {
    total: history.length,
    verified,
    rejected: history.length - verified,
    averageConfidence: totalConfidence / history.length,
    accuracy: totalAccuracy / history.length,
    passRate: verified / history.length
  };
}

export async function verifyBatch(env, items, options = {}) {
  const results = [];
  
  for (const item of items) {
    const result = await verifyTruth(env, item.query, item.result, options);
    results.push({
      query: item.query,
      ...result
    });
    
    if (options.delay) {
      await new Promise(r => setTimeout(r, options.delay));
    }
  }
  
  const verified = results.filter(r => r.verified).length;
  
  return {
    results,
    summary: {
      total: results.length,
      verified,
      rejected: results.length - verified,
      passRate: verified / results.length
    }
  };
}
