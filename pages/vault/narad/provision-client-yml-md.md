name: Provision Client

# ─────────────────────────────────────────────────────────────────────────────
# Triggered by: vishwakarma worker → POST /api/provision
#
# Payload shape (from worker dispatch):
#   clientId          string   — auto-generated project name (e.g. pcloud-deepakshar-a3f2)
#   email             string
#   companyName       string
#   adminPassword     string   — pre-generated by worker
#   storageSize       number   — 50 | 100 | 150
#   region            string   — ap-hyderabad-1 | ap-mumbai-1 etc.
#   ociTenancyOcid    string   — base64 or raw
#   ociUserOcid       string
#   ociFingerprint    string
#   ociPrivateKey     string   — always base64-encoded PEM from worker
#   useCaseId         string   — personal-cloud | small-business | ai-automation | etc.
#   projectName       string   — same as clientId (canonical name)
#   services          string   — JSON array: ["vm","nextcloud","pihole","tailscale"]
# ─────────────────────────────────────────────────────────────────────────────

on:
  repository_dispatch:
    types: [provision-client]

env:
  FORCE_JAVASCRIPT_ACTIONS_TO_NODE24: 'true'
  NODE_OPTIONS: '--no-deprecation'
  CALLBACK_BASE: 'https://vishwakarma-ops.pages.dev'
  TF_WORKSPACE: 'client-provisioning'
  TF_ORG: 'd-cloud'

jobs:
  provision:
    runs-on: ubuntu-latest
    timeout-minutes: 45

    defaults:
      run:
        working-directory: terraform

    steps:

      # ── 1. Checkout ────────────────────────────────────────────────
      - name: Checkout
        uses: actions/checkout@v4

      # ── 2. Terraform setup ─────────────────────────────────────────
      - name: Setup Terraform
        uses: hashicorp/setup-terraform@v3
        with:
          terraform_version: 1.6.6
          cli_config_credentials_hostname: app.terraform.io
          cli_config_credentials_token: ${{ secrets.TF_API_TOKEN }}

      # ── 3. Extract and mask all inputs ─────────────────────────────
      - name: Extract Client Info
        env:
          GH_TOKEN: ${{ github.token }}
        run: |
          CLIENT_ID="${{ github.event.client_payload.clientId }}"
          CLIENT_EMAIL="${{ github.event.client_payload.email }}"
          COMPANY_NAME="${{ github.event.client_payload.companyName }}"
          OCI_REGION="${{ github.event.client_payload.region || 'ap-hyderabad-1' }}"
          ADMIN_PASS="${{ github.event.client_payload.adminPassword }}"
          STORAGE_SIZE="${{ github.event.client_payload.storageSize || 150 }}"
          USE_CASE="${{ github.event.client_payload.useCaseId || 'personal-cloud' }}"
          PROJECT_NAME="${{ github.event.client_payload.projectName || github.event.client_payload.clientId }}"
          SERVICES_JSON="${{ github.event.client_payload.services || '[]' }}"

          # OCI credentials — prefer secrets over payload for security
          OCI_TENANCY="${{ secrets.OCI_TENANCY_OCID || github.event.client_payload.ociTenancyOcid }}"
          OCI_USER="${{ secrets.OCI_USER_OCID || github.event.client_payload.ociUserOcid }}"
          OCI_FINGER="${{ secrets.OCI_FINGERPRINT || github.event.client_payload.ociFingerprint }}"
          OCI_KEY_RAW="${{ secrets.OCI_PRIVATE_KEY || github.event.client_payload.ociPrivateKey }}"

          # Mask everything sensitive before any logging
          echo "::add-mask::$CLIENT_ID"
          echo "::add-mask::$CLIENT_EMAIL"
          echo "::add-mask::$ADMIN_PASS"
          echo "::add-mask::$OCI_TENANCY"
          echo "::add-mask::$OCI_USER"
          echo "::add-mask::$OCI_FINGER"
          echo "::add-mask::$OCI_KEY_RAW"

          # ── Derive service flags from services JSON array ──────────
          # jq parses the JSON array and checks for each service id.
          # Default false if services array is empty or service absent.
          INSTALL_NEXTCLOUD=$(echo "$SERVICES_JSON" | jq 'map(select(. == "nextcloud")) | length > 0')
          INSTALL_PIHOLE=$(echo "$SERVICES_JSON" | jq 'map(select(. == "pihole")) | length > 0')
          INSTALL_TAILSCALE=$(echo "$SERVICES_JSON" | jq 'map(select(. == "tailscale")) | length > 0')
          INSTALL_VAULTWARDEN=$(echo "$SERVICES_JSON" | jq 'map(select(. == "vaultwarden")) | length > 0')
          INSTALL_N8N=$(echo "$SERVICES_JSON" | jq 'map(select(. == "n8n")) | length > 0')
          INSTALL_UPTIME_KUMA=$(echo "$SERVICES_JSON" | jq 'map(select(. == "uptime_kuma")) | length > 0')

          # ── Validate required fields ───────────────────────────────
          ERROR=0
          [ -z "$CLIENT_ID" ]   && echo "::error::clientId is missing"    && ERROR=1
          [ -z "$CLIENT_EMAIL" ] && echo "::error::email is missing"       && ERROR=1
          [ -z "${{ secrets.TF_API_TOKEN }}" ] && echo "::error::TF_API_TOKEN secret not set" && ERROR=1

          if [ -z "$OCI_TENANCY" ] || [ -z "$OCI_USER" ] || [ -z "$OCI_FINGER" ] || [ -z "$OCI_KEY_RAW" ]; then
            echo "::notice::Missing OCI credentials. Assuming serverless provision request. Skipping workflow safely."
            gh run cancel ${{ github.run_id }}
            sleep 60
            exit 0
          fi

          [ $ERROR -eq 1 ] && exit 1
          echo "✅ All required inputs present"

          # ── Write to GITHUB_ENV ────────────────────────────────────
          {
            echo "CLIENT_ID=$CLIENT_ID"
            echo "CLIENT_EMAIL=$CLIENT_EMAIL"
            echo "COMPANY_NAME=$COMPANY_NAME"
            echo "OCI_REGION=$OCI_REGION"
            echo "ADMIN_PASSWORD=$ADMIN_PASS"
            echo "STORAGE_SIZE=$STORAGE_SIZE"
            echo "USE_CASE=$USE_CASE"
            echo "PROJECT_NAME=$PROJECT_NAME"
            echo "SERVICES_JSON=$SERVICES_JSON"
            echo "OCI_TENANCY=$OCI_TENANCY"
            echo "OCI_USER=$OCI_USER"
            echo "OCI_FINGER=$OCI_FINGER"
            echo "OCI_KEY_RAW=$OCI_KEY_RAW"
            echo "INSTALL_NEXTCLOUD=$INSTALL_NEXTCLOUD"
            echo "INSTALL_PIHOLE=$INSTALL_PIHOLE"
            echo "INSTALL_TAILSCALE=$INSTALL_TAILSCALE"
            echo "INSTALL_VAULTWARDEN=$INSTALL_VAULTWARDEN"
            echo "INSTALL_N8N=$INSTALL_N8N"
            echo "INSTALL_UPTIME_KUMA=$INSTALL_UPTIME_KUMA"
          } >> "$GITHUB_ENV"

      # ── 4. Decode and validate OCI PEM ────────────────────────────
      - name: Decode OCI Private Key
        env:
          OCI_KEY: ${{ env.OCI_KEY_RAW }}
        run: |
          if echo "$OCI_KEY" | grep -q "BEGIN"; then
            # Raw PEM — came from GitHub secret. Normalise newlines via base64 round-trip.
            printf '%s' "$OCI_KEY" | base64 -w 0 | base64 -d > /tmp/oci_api_key.pem
          else
            # Already base64 — standard path from worker dispatch
            printf '%s' "$OCI_KEY" | base64 -d > /tmp/oci_api_key.pem
          fi
          chmod 600 /tmp/oci_api_key.pem
          head -1 /tmp/oci_api_key.pem | grep -q "BEGIN" || { echo "::error::Invalid PEM — missing header"; exit 1; }
          LINES=$(wc -l < /tmp/oci_api_key.pem)
          [ "$LINES" -lt 10 ] && { echo "::error::PEM too short ($LINES lines) — likely corrupted"; exit 1; }
          echo "✅ OCI key decoded: $LINES lines"

      # ── 5. Notify worker: workflow started ────────────────────────
      - name: Log — Workflow Started
        run: |
          curl -sf -X PATCH \
            "${{ env.CALLBACK_BASE }}/api/clients/${{ env.CLIENT_ID }}/logs" \
            -H "Content-Type: application/json" \
            -d "{\"level\":\"info\",\"message\":\"📦 Provisioning started — use case: ${{ env.USE_CASE }}, services: ${{ env.SERVICES_JSON }}\"}" || true

      # ── 6. Set Terraform Cloud workspace variables ─────────────────
      - name: Set Terraform Cloud Variables
        env:
          TF_API_TOKEN: ${{ secrets.TF_API_TOKEN }}
        run: |
          WORKSPACE_RESPONSE=$(curl -sf \
            -H "Authorization: Bearer $TF_API_TOKEN" \
            -H "Content-Type: application/vnd.api+json" \
            "https://app.terraform.io/api/v2/organizations/${{ env.TF_ORG }}/workspaces/${{ env.TF_WORKSPACE }}")

          WORKSPACE_ID=$(echo "$WORKSPACE_RESPONSE" | jq -r '.data.id')
          [ -z "$WORKSPACE_ID" ] || [ "$WORKSPACE_ID" = "null" ] && {
            echo "::error::Failed to get workspace ID"
            echo "Response: $WORKSPACE_RESPONSE"
            exit 1
          }
          echo "✅ Workspace: $WORKSPACE_ID"

          PRIVATE_KEY=$(cat /tmp/oci_api_key.pem)

          upsert_var() {
            local KEY=$1 VALUE=$2 SENSITIVE=$3
            EXISTING=$(curl -sf \
              -H "Authorization: Bearer $TF_API_TOKEN" \
              -H "Content-Type: application/vnd.api+json" \
              "https://app.terraform.io/api/v2/workspaces/${WORKSPACE_ID}/vars" \
              | jq -r --arg k "$KEY" '.data[] | select(.attributes.key == $k) | .id')
            BODY=$(jq -n \
              --arg key "$KEY" --arg value "$VALUE" --argjson sensitive "$SENSITIVE" \
              '{data:{type:"vars",attributes:{key:$key,value:$value,sensitive:$sensitive,category:"terraform"}}}')
            if [ -n "$EXISTING" ]; then
              curl -sf -X PATCH \
                -H "Authorization: Bearer $TF_API_TOKEN" \
                -H "Content-Type: application/vnd.api+json" \
                "https://app.terraform.io/api/v2/workspaces/${WORKSPACE_ID}/vars/${EXISTING}" \
                -d "$BODY" > /dev/null && echo "  Updated: $KEY"
            else
              curl -sf -X POST \
                -H "Authorization: Bearer $TF_API_TOKEN" \
                -H "Content-Type: application/vnd.api+json" \
                "https://app.terraform.io/api/v2/workspaces/${WORKSPACE_ID}/vars" \
                -d "$BODY" > /dev/null && echo "  Created: $KEY"
            fi
          }

          # OCI credentials
          upsert_var "tenancy_ocid"        "${{ env.OCI_TENANCY }}"  true
          upsert_var "user_ocid"           "${{ env.OCI_USER }}"     true
          upsert_var "fingerprint"         "${{ env.OCI_FINGER }}"   true
          upsert_var "private_key"         "$PRIVATE_KEY"            true
          upsert_var "region"              "${{ env.OCI_REGION }}"   false
          upsert_var "tailscale_auth_key"  "${{ secrets.TAILSCALE_AUTH_KEY }}" true

      # ── 7. Write tfvars file ───────────────────────────────────────
      # CRITICAL: No leading whitespace on values. The heredoc indent
      # in the previous version caused HCL parse errors.
      - name: Create TFVars
        run: |
          mkdir -p ../clients
          cat > "../clients/${{ env.CLIENT_ID }}.tfvars" << TFEOF
          client_name              = "${{ env.CLIENT_ID }}"
          compartment_ocid         = ""
          shape                    = "VM.Standard.E2.1.Micro"
          volume_size              = ${{ env.STORAGE_SIZE }}
          nextcloud_admin_password = "${{ env.ADMIN_PASSWORD }}"
          nextcloud_admin_email    = "${{ env.CLIENT_EMAIL }}"
          nextcloud_version        = "28"
          install_nextcloud        = ${{ env.INSTALL_NEXTCLOUD }}
          install_pihole           = ${{ env.INSTALL_PIHOLE }}
          install_tailscale        = ${{ env.INSTALL_TAILSCALE }}
          install_vaultwarden      = ${{ env.INSTALL_VAULTWARDEN }}
          install_n8n              = ${{ env.INSTALL_N8N }}
          install_uptime_kuma      = ${{ env.INSTALL_UPTIME_KUMA }}
          project_name             = "${{ env.PROJECT_NAME }}"
          use_case_id              = "${{ env.USE_CASE }}"
          TFEOF

          # Fix: strip leading whitespace that heredoc indentation adds
          sed -i 's/^[[:space:]]*//' "../clients/${{ env.CLIENT_ID }}.tfvars"

          echo "✅ TFVars written:"
          cat "../clients/${{ env.CLIENT_ID }}.tfvars"

      # ── 8. Terraform init ──────────────────────────────────────────
      - name: Log — Terraform Init
        run: |
          curl -sf -X PATCH \
            "${{ env.CALLBACK_BASE }}/api/clients/${{ env.CLIENT_ID }}/logs" \
            -H "Content-Type: application/json" \
            -d '{"level":"info","message":"🔧 Terraform initialising..."}' || true

      - name: Terraform Init
        env:
          TF_API_TOKEN: ${{ secrets.TF_API_TOKEN }}
        run: terraform init

      # ── 9. Terraform apply with AD retry ──────────────────────────
      - name: Log — Planning Infrastructure
        run: |
          curl -sf -X PATCH \
            "${{ env.CALLBACK_BASE }}/api/clients/${{ env.CLIENT_ID }}/logs" \
            -H "Content-Type: application/json" \
            -d '{"level":"info","message":"📋 Planning OCI infrastructure..."}' || true

      - name: Terraform Apply (AD retry)
        id: tf_apply
        env:
          TF_API_TOKEN: ${{ secrets.TF_API_TOKEN }}
        run: |
          SUCCESS=false
          for RETRY in 1 2 3; do
            AD_INDEX=$((RETRY - 1))
            echo "=== Attempt $RETRY/3 (AD index $AD_INDEX) ==="

            curl -sf -X PATCH \
              "${{ env.CALLBACK_BASE }}/api/clients/${{ env.CLIENT_ID }}/logs" \
              -H "Content-Type: application/json" \
              -d "{\"level\":\"info\",\"message\":\"🖥️ Creating VM (attempt $RETRY/3, AZ $((AD_INDEX+1)))...\"}" || true

            terraform plan \
              -var-file="../clients/${{ env.CLIENT_ID }}.tfvars" \
              -var="preferred_ad_index=${AD_INDEX}" \
              -out=tfplan

            if terraform apply -auto-approve tfplan; then
              SUCCESS=true
              break
            fi

            if [ $RETRY -lt 3 ]; then
              curl -sf -X PATCH \
                "${{ env.CALLBACK_BASE }}/api/clients/${{ env.CLIENT_ID }}/logs" \
                -H "Content-Type: application/json" \
                -d "{\"level\":\"warn\",\"message\":\"⚠️ AZ $((AD_INDEX+1)) unavailable — trying AZ $((AD_INDEX+2))...\"}" || true
              sleep 30
              terraform destroy -auto-approve \
                -var="preferred_ad_index=${AD_INDEX}" \
                -var-file="../clients/${{ env.CLIENT_ID }}.tfvars" || true
            fi
          done

          [ "$SUCCESS" = false ] && { echo "::error::All 3 AZ attempts failed"; exit 1; }

      # ── 10. Capture Terraform outputs ─────────────────────────────
      - name: Get Terraform Outputs
        id: tf
        env:
          TF_API_TOKEN: ${{ secrets.TF_API_TOKEN }}
        run: |
          PUBLIC_IP=$(terraform output -raw public_ip)
          SSH_KEY=$(terraform output -raw ssh_private_key)
          PIHOLE_PASS=$(terraform output -raw pihole_admin_password)
          DOMAIN=$(terraform output -raw domain)
          NC_URL=$(terraform output -raw nextcloud_url)

          echo "public_ip=${PUBLIC_IP}"                              >> "$GITHUB_OUTPUT"
          echo "nextcloud_url=${NC_URL}"                            >> "$GITHUB_OUTPUT"
          echo "domain=${DOMAIN}"                                   >> "$GITHUB_OUTPUT"
          echo "pihole_password=${PIHOLE_PASS}"                     >> "$GITHUB_OUTPUT"
          echo "ssh_key_b64=$(echo "$SSH_KEY" | base64 -w 0)"      >> "$GITHUB_OUTPUT"

          echo "$SSH_KEY" > /tmp/ssh_key_${{ env.CLIENT_ID }}.pem
          chmod 600 /tmp/ssh_key_${{ env.CLIENT_ID }}.pem

      # ── 11. Wait for cloud-init ────────────────────────────────────
      - name: Log — VM Live, Installing Services
        run: |
          curl -sf -X PATCH \
            "${{ env.CALLBACK_BASE }}/api/clients/${{ env.CLIENT_ID }}/logs" \
            -H "Content-Type: application/json" \
            -d "{\"level\":\"info\",\"message\":\"🌐 VM live at ${{ steps.tf.outputs.public_ip }} — installing selected services...\"}" || true

      - name: Wait for Cloud-Init
        run: |
          echo "Waiting 2 min for VM boot..."
          sleep 120

          for i in $(seq 1 40); do
            STATUS=$(ssh -i /tmp/ssh_key_${{ env.CLIENT_ID }}.pem \
              -o StrictHostKeyChecking=no -o ConnectTimeout=8 \
              ubuntu@${{ steps.tf.outputs.public_ip }} \
              "cat /opt/vishwakarma/status/status.txt 2>/dev/null || echo 'pending'" 2>/dev/null || echo "unreachable")

            echo "Attempt $i/40 — status: $STATUS"

            [ "$STATUS" = "deployed" ] && { echo "✅ Cloud-init complete"; break; }

            if [ $i -eq 40 ]; then
              echo "⚠️ Timed out — services may still be starting"
            else
              # Log progress messages from cloud-init back to the UI
              LAST_LOG=$(ssh -i /tmp/ssh_key_${{ env.CLIENT_ID }}.pem \
                -o StrictHostKeyChecking=no -o ConnectTimeout=8 \
                ubuntu@${{ steps.tf.outputs.public_ip }} \
                "tail -1 /var/log/cloud-deploy.log 2>/dev/null" 2>/dev/null || echo "")
              [ -n "$LAST_LOG" ] && curl -sf -X PATCH \
                "${{ env.CALLBACK_BASE }}/api/clients/${{ env.CLIENT_ID }}/logs" \
                -H "Content-Type: application/json" \
                -d "{\"level\":\"info\",\"message\":\"$(echo "$LAST_LOG" | sed 's/"/\\"/g')\"}" || true
            fi

            sleep 15
          done

      # ── 12. Collect installed service URLs from VM ─────────────────
      - name: Collect Service URLs
        id: svc_urls
        run: |
          IP="${{ steps.tf.outputs.public_ip }}"
          SSH="ssh -i /tmp/ssh_key_${{ env.CLIENT_ID }}.pem -o StrictHostKeyChecking=no -o ConnectTimeout=10 ubuntu@${IP}"

          # Collect all service access details in one SSH call
          SERVICE_INFO=$($SSH "cat /opt/vishwakarma/service-urls.json 2>/dev/null || echo '{}'" 2>/dev/null || echo '{}')
          echo "service_info_b64=$(echo "$SERVICE_INFO" | base64 -w 0)" >> "$GITHUB_OUTPUT"

          # Also get Pi-hole setup guide
          PIHOLE_GUIDE=$($SSH "cat /opt/vishwakarma/pihole-setup-guide.txt 2>/dev/null || echo ''" 2>/dev/null || echo "")
          echo "pihole_guide_b64=$(echo "$PIHOLE_GUIDE" | base64 -w 0)" >> "$GITHUB_OUTPUT"

      # ── 13. Mark complete and send credentials ─────────────────────
      - name: Log — Services Ready
        run: |
          curl -sf -X PATCH \
            "${{ env.CALLBACK_BASE }}/api/clients/${{ env.CLIENT_ID }}/logs" \
            -H "Content-Type: application/json" \
            -d '{"level":"success","message":"✅ All services deployed and ready!"}' || true

      - name: Mark Provisioning Complete
        run: |
          SSH_KEY=$(echo "${{ steps.tf.outputs.ssh_key_b64 }}" | base64 -d)
          PIHOLE_GUIDE=$(echo "${{ steps.svc_urls.outputs.pihole_guide_b64 }}" | base64 -d)
          SERVICE_INFO=$(echo "${{ steps.svc_urls.outputs.service_info_b64 }}" | base64 -d)

          curl -sf -X POST \
            "${{ env.CALLBACK_BASE }}/api/clients/${{ env.CLIENT_ID }}/complete" \
            -H "Content-Type: application/json" \
            -d "$(jq -n \
              --arg status            "deployed" \
              --arg vmPublicIp        "${{ steps.tf.outputs.public_ip }}" \
              --arg nextcloudUrl      "${{ steps.tf.outputs.nextcloud_url }}" \
              --arg nextcloudAdminPassword "${{ env.ADMIN_PASSWORD }}" \
              --arg domain            "${{ steps.tf.outputs.domain }}" \
              --arg piholePassword    "${{ steps.tf.outputs.pihole_password }}" \
              --arg tailscaleHostname "${{ env.CLIENT_ID }}-server" \
              --arg piholeGuide       "$PIHOLE_GUIDE" \
              --arg sshKey            "$SSH_KEY" \
              --argjson serviceInfo   "$SERVICE_INFO" \
              --arg useCaseId         "${{ env.USE_CASE }}" \
              --arg services          "${{ env.SERVICES_JSON }}" \
              '{
                status:                  $status,
                vmPublicIp:              $vmPublicIp,
                nextcloudUrl:            $nextcloudUrl,
                nextcloudAdminPassword:  $nextcloudAdminPassword,
                domain:                  $domain,
                piholePassword:          $piholePassword,
                tailscaleHostname:       $tailscaleHostname,
                piholeGuide:             $piholeGuide,
                sshKey:                  $sshKey,
                serviceInfo:             $serviceInfo,
                useCaseId:               $useCaseId,
                services:                $services
              }')"

      # ── 14. Always upload SSH key as artifact ──────────────────────
      - name: Upload SSH Key Artifact
        if: always()
        uses: actions/upload-artifact@v4
        with:
          name: ssh-key-${{ env.CLIENT_ID }}
          path: /tmp/ssh_key_${{ env.CLIENT_ID }}.pem
          retention-days: 7

      # ── 15. Failure handler ────────────────────────────────────────
      - name: Handle Failure
        if: failure()
        run: |
          curl -sf -X PATCH \
            "${{ env.CALLBACK_BASE }}/api/clients/${{ env.CLIENT_ID }}/logs" \
            -H "Content-Type: application/json" \
            -d '{"level":"error","message":"❌ Provisioning failed — check GitHub Actions run for details"}' || true

          curl -sf -X POST \
            "${{ env.CALLBACK_BASE }}/api/clients/${{ env.CLIENT_ID }}/complete" \
            -H "Content-Type: application/json" \
            -d '{"status":"failed"}' || true
